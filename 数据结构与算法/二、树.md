#### 根据中序遍历和后序遍历求先序遍历
```
class Main{
    static char[] midOrder;
    static char[] lastOrder;
    public static void main(String args[]) {
        Scanner scan = new Scanner(System.in);
        int n;
        while (scan.hasNextLine()) {
            String midString = scan.nextLine();
            String lastString = scan.nextLine();
            midOrder = midString.toCharArray();
            lastOrder = lastString.toCharArray();
            getPreOrder(0, midOrder.length, 0, lastOrder.length);
        }
    }
    private static void getPreOrder(int midLeft, int midRight, int lastLeft, int lastRight) {
        if (midLeft >= midRight) return;
        if (lastLeft >= lastRight) return;
        char head = lastOrder[lastRight - 1];
        System.out.print(head);
        for (int i = midLeft; i < midRight; i++) {
            if (midOrder[i] == head) {
                int len = i - midLeft;
                getPreOrder(midLeft, i, lastLeft, lastLeft + len);
                getPreOrder(i + 1, midRight, lastLeft + len, lastRight - 1);
                break;
            }
        }
    }
}
```
